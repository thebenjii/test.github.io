// Global variables let stars = []; function setup() { createCanvas(windowWidth, windowHeight); generateStars(200); // Adjust the number as desired } function draw() { background(0); drawStars(); } function generateStars(count) { for (let i = 0; i < count; i++) { let star = { x: random(width), y: random(height), size: random(1, 3), brightness: random(150, 255) }; stars.push(star); } } function drawStars() { noStroke(); for (let star of stars) { fill(star.brightness); ellipse(star.x, star.y, star.size); } } function windowResized() { resizeCanvas(windowWidth, windowHeight); stars = []; generateStars(200); }
